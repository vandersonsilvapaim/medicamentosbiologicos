import {
  Component, OnInit, OnDestroy, HostListener,
  Output, EventEmitter, Input
} from '@angular/core';
import { FormGroup } from '@angular/forms';
import { Router } from '@angular/router';
import { AlertService } from '../../shared/services/alert.service';
import { LogService } from '../../shared/services/log.service';


@Component({
  selector: 'dispensacao-form',
  templateUrl: './dispensacao-form.component.html',
  styleUrls: ['./dispensacao-form.component.scss'],
  preserveWhitespaces: false
})
export class DispensacaoFormComponent implements OnInit, OnDestroy {

  public isLive = false;
  public fb;
  public _fb;
  public submmit;

  @Output() public action = new EventEmitter();

  @Input() public set forms(_forms: FormGroup) {
    if (!this.isLive) {
      return;
    }

  }

  @HostListener('mouseleave') public onMouseLeave() {
    // this.keyupNumber();
    // this.blurDate();
  }

  constructor(
    public log: LogService,
    private router: Router,
    public alert: AlertService
  ) { }

  ngOnDestroy(): void {
    this.isLive = false;
    this.fb = null;
    this._fb = null;
    this.alert.clearAll();
  }

  ngOnInit(): void {
    this.isLive = true;
  }

  /**
   * metods
   */

  public back_stepper() {
    this.action.emit({ back_stepper: true });
  }

}
